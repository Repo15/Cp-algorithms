1.Sieve of Eratosthenes 
      vector sieve_of_eratosthenes(int sieve)
      {
      vector arr(sieve + 1, 0);
      arr[0] = 1;
      arr[1] = 1;
      for (int i = 2; i * i <= sieve; i++)
          for (int j = 2 * i; j <= sieve; j += i)
              arr[j] = 1;
      return arr;
      }
